## Plot Probability Density Function

density_option=True
## If Density = True, draw and return a probability density: each bin will display the bin's 
## raw count divided by the total number of counts and the bin width 
##(density = counts / (sum(counts) * np.diff(bins))), 
## so that the area under the histogram integrates to 1 (np.sum(density * np.diff(bins)) == 1).

plt.rcParams["figure.figsize"] = [4,4];
plt.hist(y, bins=xinc, histtype='bar', rwidth=0.5, alpha=1,color='dodgerblue',density=density_option);
plt.hist(x, bins=xinc, histtype='bar', rwidth=0.5, alpha=1,color='firebrick',density=density_option);
plt.xlabel('Value');
plt.ylabel('Probability Density');
plt.legend(['small', 'big'],loc='upper right',frameon=True,markerscale=0.5,framealpha=1);
